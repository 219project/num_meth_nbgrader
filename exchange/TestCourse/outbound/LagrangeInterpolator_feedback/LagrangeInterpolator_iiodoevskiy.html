import numpy as np

class LagrangeInterpolator:
    """Lagrange interpolating polynomial.
    
    Given a set of pairs ``(x_k, y_k)``, construct 
    a Lagrange polynomial ``f(x)``, such that
    
    .. math::

        f(x_k) = y_k   for k =0, ..., n-1
    
    Parameters
    ----------
    xk : array_like, shape(n,)
        Abscissas
    yk : array_like, shape(n,)
        Ordinates
    
    Attributes
    ----------
    __call__
    
    """
    def __init__(self, xk, yk):
        self.xk = np.asarray(xk, dtype=float)
        self.yk = np.asarray(yk, dtype=float)
        
    def __call__(self, x):
        """Evaluate the interpolator at a given point.
        
        Parameters
        ----------
        x : float
        
        Returns
        -------
        the value of the interpolator at ``x``.
        """
        summ = 0
        for j in range(len(self.yk)):
            chisl=1; znam=1
            for i in range(len(self.xk)):
                if i==j:
                    continue  
                else: 
                    chisl*=x-self.xk[i]
                    znam*=self.xk[j]-self.xk[i]
            summ+=self.yk[j]*chisl/znam
        return summ

def chebyshev_nodes(a, b, n):
    nodes = []
    for i in range(n):
        x = 0.5*(a+b) + 0.5*(b-a)*np.cos((2*i+1)*np.pi/(2*n))
        nodes.append(x)       
    return np.array(nodes)
for i in range(4):
    cheb = chebyshev_nodes(-2, 2, m[i])
    yy = runge_func(cheb)
    lagr = LagrangeInterpolator(cheb, yy)
    xnew = np.linspace(-2,2,100)
    ynew = lagr(xnew)
    plt.plot(cheb,yy,'o',xnew,ynew)
    plt.grid()
    plt.show()
