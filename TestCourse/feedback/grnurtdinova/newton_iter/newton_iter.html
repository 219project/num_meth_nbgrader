def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    
    xn = x0
    for niter in range(0,maxiter):
        fxn = f(xn)
        if abs(fxn) < eps:
            #print('Found solution after',n,'iterations.')
            return xn, niter
        Dfxn = fder(xn)
        xn = xn - fxn/Dfxn
    return xn, niter

def mod_newton(f, fder, x0, m, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
             c  .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    
    x = x0 - m*f(x0)/fder(x0)
    niter = 1
    while niter < maxiter:
        if abs(x-x0) < eps:
            return x, niter
        x0 = x
        x = x0 - m*f(x0)/fder(x0)
        niter += 1
    return x, niter
