def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    niter = 0
    x = x0
    for i in range(maxiter):
        x_new = x - f(x)/fder(x)
        niter += 1
        if abs(x - x_new) <= eps:
            break
        x = x_new
    return x, niter

def newton_system(F, jcb, X0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    F : callable
        - ,    .
    jcb : callable
         `f`.
    X0 : array-like of floats, shape (n,)
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    X : array-like of floats, shape (n,)
           .
    niter : int
         .
    """
    niter = 0
    X = np.copy(X0)
    for i in range(maxiter):
        d_X = np.linalg.solve(jcb(X), -F(X))
        X_new = X + d_X
        if np.linalg.norm(X - X_new) <= eps:
            niter = i+1
            break
        X = X_new
    return X, niter
