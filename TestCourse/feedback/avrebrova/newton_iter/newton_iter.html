def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    x0 = x0
    err = eps + 1
    niter = 0
    while err >= eps and niter < maxiter:
      niter += 1
      x = x0 - f(x0)/fder(x0)
      err = abs(x - x0)
      x0 = x
    return x0, niter

def newton_system(F, jcb, X0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    F : callable
        - ,    .
    jcb : callable
         `f`.
    X0 : array-like of floats, shape (n,)
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    X : array-like of floats, shape (n,)
           .
    niter : int
         .
    """
    X0 = X0
    err = eps + 1
    niter = 0
    while err >= eps and niter < maxiter:
      niter += 1
      x = X0 - np.linalg.inv(jcb(X0)) @ F(X0)
      err = np.linalg.norm(abs(x - X0))
      X0 = x
    return X0, niter
