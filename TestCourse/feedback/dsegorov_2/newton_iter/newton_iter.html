def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    x = 0
    for i in range(maxiter):
        x = x0 - f(x0) / fder(x0)
        if abs(x - x0) < eps:
            return x, i
        x0 = x
    return x0, maxiter

def newton_system(F, jcb, X0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    F : callable
        - ,    .
    jcb : callable
         `f`.
    X0 : array-like of floats, shape (n,)
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    X : array-like of floats, shape (n,)
           .
    niter : int
         .
    """

    from numpy.linalg import norm, solve

    xn = np.asarray(X0)
    for i in range(maxiter):
        xn1 = solve(jcb(xn), -F(xn)) + xn
        if norm(xn1 - xn) < eps:
            return xn1, i
        xn = xn1
    return xn, maxiter
